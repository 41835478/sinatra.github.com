571:       def use_in_file_templates!
572:         ignore = [/lib\/sinatra.*\.rb/, /\(.*\)/, /rubygems\/custom_require\.rb/]
573:         file = caller.
574:           map  { |line| line.sub(/:\d+.*$/, '') }.
